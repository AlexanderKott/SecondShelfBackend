SELECT public.applications.* FROM public.applications
INNER JOIN public.things
ON public.things.id = public.applications.thing_id
INNER JOIN public.users
ON public.things.user_id = public.users.id
WHERE public.users.id = '4706beb9-76ef-4f42-a9d2-454f381978e3'
AND public.applications.delete_date IS NULL
-- AND public.things.id = 'aa79ec8e-10b8-4c18-af0d-472d13210619'  -- на конкретную вещь
-- AND public.applications.status = 1 -- вернуть заявки с нужным статусом
 

--Получить по id вещи её статус участия в возможном движении по воркфоу (участии в заявках). 
WITH thing_lookup AS
    (SELECT * FROM public.things 
	 LEFT JOIN public.applications ON public.things.id = public.applications.thing_id
	WHERE public.applications.thing_id = 'aa79ec8e-10b8-4c18-af0d-472d13210619'),
	 statuses_temp AS 
  (SELECT CASE
  WHEN status is NULL THEN '1'
  ELSE status
END AS status
FROM thing_lookup)
SELECT max(status)
FROM statuses_temp;

SELECT COALESCE(public.applications.status, '0') AS status, public.things.* FROM public.things
LEFT JOIN public.applications ON public.things.id = public.applications.thing_id
WHERE public.things.delete_date is NULL
--AND COALESCE(public.applications.status, '0') IN (0, 1, 2) -- в общую ленту идут только такие статусы
--AND public.things.name ILIKE CONCAT('%', '', '%')
--AND public.things.category = 1
--AND public.things.location = 3
--AND public.things.user_id = '6d42e365-be3e-4b9b-bdbf-9c3767b462eb' -- найти для определенного дарителя
ORDER BY add_date;
LIMIT 20 OFFSET 0 -- limit = page_size = 20, offset = page_size * page_number(starts from zero)
